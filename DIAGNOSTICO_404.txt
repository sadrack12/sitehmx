COMANDOS PARA DIAGNOSTICAR ERRO 404:

# 1. Verificar estrutura
cd ~/public_html/api
pwd
ls -la

# 2. Verificar se public/ existe
ls -la public/

# 3. Verificar se index.php existe
ls -la public/index.php

# 4. Verificar .htaccess
ls -la public/.htaccess

# 5. Ver rotas disponíveis
php artisan route:list | grep public

# 6. Testar acesso direto ao index.php
php public/index.php

# 7. Ver logs
tail -20 storage/logs/laravel.log
